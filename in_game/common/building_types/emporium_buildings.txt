# Emporium Mod - Buildings

# Level 1
emp_trade_hub = {
    is_foreign = no

    graphical_tags = { emp_trade_hub }
    pop_type = burghers
    max_levels = 1
    employment_size = 2.5

    category = trade_category

    city = yes

    build_time = 720
    price = emp_trade_hub_price
    expensive = yes

    country_potential = {
        # Rank: Count or above
        country_rank_level >= 1
        # Strict 1-per-country: no emporium of ANY level anywhere in the country
        NOT = {
            any_scope_state = {
                any_scope_building = {
                    OR = {
                        is_building_type = emp_trade_hub
                        is_building_type = emp_grand_hub
                        is_building_type = emp_royal_hub
                        is_building_type = emp_imperial_hub
                    }
                }
            }
        }
    }

    location_potential = {
        is_capital = yes
        is_market_center = yes
        OR = {
            is_coastal = yes
            has_river = yes
        }
        # Also block building a second one in the same city slot
        NOT = {
            has_building = emp_trade_hub
        }
    }

    modifier = {
        local_market_attractiveness = 0.05
        local_merchant_power = 25
        local_merchant_capacity = 25
        local_trades_per_burgher = 0.25
    }

    capital_country_modifier = {
        trade_range_modifier = 0.25
    }

    market_center_modifier = {
        local_market_attractiveness = 0.05
        local_trade_center_power = 0.05
    }

    unique_production_methods = {
        emp_trade_hub_maintenance = {
            fine_cloth = 1.25
            paper = 1.25
            cloth = 1.25
            beeswax = 0.5
            jewelry = 1.25
            category = building_maintenance
        }
    }
    construction_demand = emporium_l1_construction

    on_built = {
        owner = {
            trigger_event_silently = { id = emporium.1 days = 360 }
        }
    }
}
